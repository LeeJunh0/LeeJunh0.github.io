# 풀이

채워진 배열에 입력 값에 해당하는 열 전체를 바꿔주면 되는 문제다.

```c++
int main()
{
    int arrays[5][5] =
    {
        { 21, 16, 11, 6, 1 },
        { 22, 17, 12, 7, 2 },
        { 23, 18, 13, 8, 3 },
        { 24, 19, 14, 9, 4 },
        { 25, 20, 15, 10, 5 }
    };

    int input;
    std::cin >> input;

    for(int i = 0; i < 5; i++)
        arrays[input][i] = input;

    for(int i = 0; i < 5; i++)
    {
        for(int j = 0; j < 5; j++)
            std::cout << arrays[i][j] << " ";

        std::cout << std::endl;
    }

    // 입력 : 4
    // 결과
    // 21 16 11 6 1
    // 22 17 12 7 2
    // 23 18 13 8 3
    // 24 19 14 9 4
    // 4 4 4 4 4

    return 0;
}
```
