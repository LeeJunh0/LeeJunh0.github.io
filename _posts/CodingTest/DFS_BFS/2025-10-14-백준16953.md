# 풀이

```c++
#include <iostream>
#include <string>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int a, b, count = 0;
	cin >> a >> b;

	while (b > a)
	{
		if (b % 2 == 0)
			b /= 2;
		else
		{
			if (b % 10 == 1)
				b /= 10;
			else
				break;
		}

		count++;
	}

	if (a == b)
		cout << count + 1 << '\n';
	else
		cout << -1 << '\n';

	return 0;
}
```
