# 풀이

```c++
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int n, m;
	cin >> n;
	cin >> m;
	vector<int> vec(n);
	for (int i = 0; i < n; i++)
		cin >> vec[i];
	
	sort(vec.begin(), vec.end());

	int count = 0;
	int left = 0;
	int right = n - 1;
	int curSum = 0;
	while (left != right)
	{
		curSum = vec[left] + vec[right];
		if (curSum == m)
		{
			count++;
			left++;
		}
		else if (curSum < m)
			left++;
		else
			right--;
	}

	cout << count << '\n';
	return 0;
}
```
