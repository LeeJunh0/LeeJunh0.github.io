# 풀이

```c++
#include <iostream>
#include <vector>
#define ull unsigned long long
#define Matrix vector<vector<ull>>
#define Divide 1000000007

using namespace std;

Matrix Multiply(const Matrix& a, const Matrix& b)
{
	Matrix result(2, vector<ull>(2));
	result[0][0] = (a[0][0] * b[0][0] + a[0][1] * b[1][0]) % Divide;
	result[0][1] = (a[0][0] * b[0][1] + a[0][1] * b[1][1]) % Divide;
	result[1][0] = (a[1][0] * b[0][0] + a[1][1] * b[1][0]) % Divide;
	result[1][1] = (a[1][0] * b[0][1] + a[1][1] * b[1][1]) % Divide;

	return result;
}

Matrix Power(Matrix base, long long n)
{
	Matrix result = { {1, 0}, {0, 1} };
	while (n)
	{
		if (n % 2 == 1)
			result = Multiply(result, base);

		base = Multiply(base, base);
		n = n / 2;
	}

	return result;
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	ull n;
	cin >> n;

	Matrix base = { {1, 1},{1, 0} };
	Matrix result = Power(base, n - 1);
	cout << result[0][0] << '\n';
	return 0;
}
```
