# 풀이

```c++
#include <bits/stdc++.h>
#include <unordered_map>
#include <algorithm>

using namespace std;

int ground[501][501];

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int y, x, invenCnt;
	cin >> y >> x >> invenCnt;

	double sum = 0;
	int maxHeight = INT_MIN;
	for (int i = 0; i < y; ++i)
	{
		for (int j = 0; j < x; ++j)
		{
			cin >> ground[i][j];
			maxHeight = max(maxHeight, ground[i][j]);
		}
	}

	vector<int> timeVec(maxHeight + 1);
	vector<int> minTimeVec;
	int minTime = INT_MAX;

	for (int cur = 0; cur <= maxHeight; ++cur)
	{
		pair<int, int> averageCnt = { 0, 0 };
		int curInven = invenCnt;
		for (int i = 0; i < y; ++i)
		{
			for (int j = 0; j < x; ++j)
			{
				if (ground[i][j] > cur)
				{
					curInven += ground[i][j] - cur;
					averageCnt.first += ground[i][j] - cur;
				}
				else if (ground[i][j] < cur)
				{
					averageCnt.second += cur - ground[i][j];
				}
			}
		}
		
		if (averageCnt.second > curInven)
		{
			timeVec[cur] = INT_MAX;
			continue;
		}

		int averageTime = averageCnt.first * 2 + averageCnt.second;
		timeVec[cur] = averageTime;
		minTime = min(minTime, averageTime);
	}	

	for (int i = 0; i <= maxHeight; ++i)
	{
		if (minTime == timeVec[i])
			minTimeVec.push_back(i);
	}

	cout << minTime << " " << minTimeVec.back() << '\n';
	return 0;
}
```

# 후기

땅을 평평하게 만드는 문제였다.

처음에는 최소, 최대, 평균 높이를 구해 이중 가장 적게 걸리는 높이로 하고자 했는데 그렇게 접근하면 안됬다..ㅋㅋ

무식하게 높이 0에서 땅에서 제일 높은 블록의 Height까지 시간을 계산하고

그 중 가장 적은 시간이 걸리는 높이를 찾아 리턴 하면 되는데.

그 시간이 똑같은 높이가 여러개 있다면 그 중 가장 높은 높이로 출력해야했다.

그래서 MinTime 높이들을 전부 벡터에 담아 가장 나중에 담기는 높이를 출력하는걸로 해결했다.
