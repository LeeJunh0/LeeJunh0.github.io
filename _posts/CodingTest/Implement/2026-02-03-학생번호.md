# 풀이

```c++
#include <bits/stdc++.h>
#include <unordered_map>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int result = 0;

	int len = 0;
	unordered_map<int, vector<string>> map(n);
	for (int i = 0; i < n; i++)
	{
		string str;
		cin >> str;
		len = str.size();
		bool isCheck = false;
		for (int j = 0; j < str.size(); j++)
		{
			string cur = str.substr(str.size() - 1 - j, 1 + j);
			if (find(map[j + 1].begin(), map[j + 1].end(), cur) == map[j + 1].end())
				map[j + 1].push_back(cur);
		}
	}


	for (int i = 0; i < len; i++)
	{
		if (map[i + 1].size() == n)
		{
			result = i + 1;
			break;
		}
	}

	std::cout << result << '\n';
	return 0;
}
```
