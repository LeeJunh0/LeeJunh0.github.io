# 풀이

```c++
#include <bits/stdc++.h>
#include <unordered_map>

using namespace std;

vector<string> strVec;

void Sum(string cube)
{
	vector<bool> visited(cube.size(), false);
	map<char, int> cubeMap;
	map<char, int> haveMap;
	for (int i = 0; i < cube.size(); i++)
	{
		cubeMap[cube[i]] = 0;
		haveMap[cube[i]]++;
	}

	for (int i = 0; i < strVec.size(); i++)
	{
		map<char, int> strMap;
		for (int j = 0; j < strVec[i].size(); j++)
			strMap[strVec[i][j]]++;

		bool isComplete = true;
		for (int j = 0; j < strVec[i].size(); j++)
		{
			if (haveMap.find(strVec[i][j]) == haveMap.end())
			{
				isComplete = false;
				break;
			}

			if (haveMap[strVec[i][j]] < strMap[strVec[i][j]])
			{
				isComplete = false;
				break;
			}
		}

		if (isComplete == true)
		{
			for (auto word : strMap)
			{
				cubeMap[word.first]++;
			}
		}	
	} 

	int maxCnt = 0;
	int minCnt = INT_MAX;
	string minCh = "";
	string maxCh = "";

	for (pair<char, int> word : cubeMap)
	{
		maxCnt = max(maxCnt, word.second);
		minCnt = min(minCnt, word.second);
	}

	for (pair<char, int> word : cubeMap)
	{
		if (word.second == maxCnt)
			maxCh.push_back(word.first);

		if (word.second == minCnt)
			minCh.push_back(word.first);
	}

	cout << minCh << " " << minCnt << " " << maxCh << " " << maxCnt << '\n';
}

int main()
{
	ios_base::sync_with_stdio(false);
	cout.tie(NULL);
	cin.tie(NULL);

	while (true)
	{
		string str;
		cin >> str;
		if (str == "-")
			break;

		strVec.push_back(str);
	}

	while (true)
	{
		string str;
		cin >> str;
		if (str == "#")
			break;

		Sum(str);
	}

	return 0;
}
```
