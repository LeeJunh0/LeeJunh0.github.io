# 풀이

```c++
#include <bits/stdc++.h>
#include <unordered_map>
#include <algorithm>

using namespace std;

pair<int, int> RotateRight(pair<int, int> curFront)
{
	if (curFront == make_pair(1, 0))
		return make_pair(0, -1);
	else if (curFront == make_pair(0, -1))
		return make_pair(-1, 0);
	else if (curFront == make_pair(-1, 0))
		return make_pair(0, 1);
	else
		return make_pair(1, 0);
}

pair<int, int> RotateLeft(pair<int, int> curFront)
{
	if (curFront == make_pair(1, 0))
		return make_pair(0, 1);
	else if (curFront == make_pair(0, 1))
		return make_pair(-1, 0);
	else if (curFront == make_pair(-1, 0))
		return make_pair(0, -1);
	else
		return make_pair(1, 0);
}

int main()
{
	ios_base::sync_with_stdio(false);
	cout.tie(NULL);
	cin.tie(NULL);

	int n;
	cin >> n;
	char map[101][101];
	fill(&map[0][0], &map[0][0] + (101 * 101), '#');
	pair<int, int> maxPos = { 50, 50 };
	pair<int, int> minPos = { 50, 50 };
	pair<int, int> front = { 1, 0 };
	pair<int, int> curPos = { 50, 50 };
	map[50][50] = '.';

	for (int i = 0; i < n; i++)
	{
		char command;
		cin >> command;

		if (command == 'L')
			front = RotateLeft(front);
		else if (command == 'R')
			front = RotateRight(front);
		else
		{
			curPos = { curPos.first + front.first, curPos.second + front.second };
			map[curPos.first][curPos.second] = '.';
			minPos = { min(minPos.first, curPos.first), min(minPos.second, curPos.second) };
			maxPos = { max(maxPos.first, curPos.first), max(maxPos.second, curPos.second) };
		}
	}

	for (int i = minPos.first; i <= maxPos.first; i++)
	{
		for (int j = minPos.second; j <= maxPos.second; j++)
			cout << map[i][j];

		cout << '\n';
	}
	return 0;
}
```

# 후기

이 문제는 기준점을 어떻게 잡아야 할지 몰라서 헤맸다..

입력에 보면 50만큼 입력이 들어온다 되어있는데 즉 한 방향으로 50만큼 갈 수도 있단 의미다.

그렇다는건 반지름 50을 잡고 101, 101 사이즈로 맵을 만들고 50, 50을 기준점으로 만들면 된다는 의미였다...(그냥 알려줘..)

그 이후는 전진방향을 우로 좌로 잘 돌리고 현재 위치에서 움직인 최소 최대 Y, X 값을 다 저장한다.

그리고 그 만큼 출력해주면 끝이다.
