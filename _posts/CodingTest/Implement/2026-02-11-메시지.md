# 풀이

```c++
#include <bits/stdc++.h>
#include <unordered_map>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cout.tie(NULL);
	cin.tie(NULL);

	int curGroup = 1;
	while (true)
	{
		int n;
		cin >> n;
		if (n == 0)
			break;

		vector<string> names(n);
		vector<vector<char>> message(n);
		for (int i = 0; i < n; i++)
		{
			cin >> names[i];
			for (int j = 0; j < n - 1; j++)
			{
				char ch;
				cin >> ch;
				message[i].push_back(ch);
			}
		}

		cout << "Group " << curGroup << '\n';
		bool isCheck = true;
		for (int i = 0; i < n; i++)
		{
			int cur = i;
			for (int j = 0; j < n - 1; j++)
			{
				if (cur - 1 < 0)
					cur = n - 1;
				else
					cur -= 1;

				if (message[i][j] == 'N')
				{
					cout << names[cur] << " was nasty about " << names[i] << '\n';
 					isCheck = false;
				}
			}
		}

		if (isCheck == true)
			cout << "Nobody was nasty" << '\n';

		curGroup++;
		cout << '\n';
	}
	return 0;
}
```

# 후기

자신에 순서에서 왼쪽으로 돌아가며 좋은 말을 해주는 건데 나쁜말을 한 녀석을 찾는 문제다.

접근 방법은 이랬다.

1. P든 N이든 일단 왼쪽으로 움직인다.

2. 현재 순서에 -1을 했을때 0보다 작은경우 끝으로 보낸다.

3. N이 나온경우 cur번째 녀석이 나쁜말을 한 녀석이므로 출력해준다. 끝
