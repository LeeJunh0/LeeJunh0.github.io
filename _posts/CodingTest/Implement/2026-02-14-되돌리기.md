# 풀이

```c++
#include <bits/stdc++.h>
#include <unordered_map>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cout.tie(NULL);
	cin.tie(NULL);

	int n;
	cin >> n;

	string type;
	string ch;
	int sec;

	vector<pair<int, string>> vec;
	vec.push_back({ 0, "" });
	for (int i = 0; i < n; i++)
	{
		cin >> type >> ch >> sec;

		if (type == "type")
		{
			string lastString = vec.back().second;
			vec.push_back({ sec, lastString + ch });
		}
		else
		{
			int index = vec.size() - 1;
			for (int j = 0; j < vec.size(); j++)
			{
				if (sec - stoi(ch) <= vec[j].first)
				{
					index = j - 1;
					break;
				}
			}
			if (index == -1)
				vec.push_back(make_pair(sec, ""));
			else
				vec.push_back(make_pair(sec, vec.at(index).second));
		}
	}

	if (vec.back().second != "")
		cout << vec.back().second << "\n";

	return 0;
}
```
