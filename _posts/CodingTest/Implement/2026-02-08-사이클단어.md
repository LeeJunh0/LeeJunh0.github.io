# 풀이

```c++
#include <bits/stdc++.h>
#include <unordered_map>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cout.tie(NULL);
	cin.tie(NULL);

	int n;
	cin >> n;
	set<string> strSet;
	int result = 0;
	for (int i = 0; i < n; i++)
	{
		string str;
		cin >> str;
		if (strSet.find(str) == strSet.end())
		{
			result++;

			for (int j = 0; j < str.size(); j++)
			{
				string cur = "";

				for (int k = 0; k < str.size(); k++)
					cur += str[(k + j) % str.size()];

				strSet.insert(cur);
			}
		}
	}

	std::cout << result << '\n';
	return 0;
}
```
