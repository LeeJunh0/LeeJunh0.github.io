# 풀이
  


```c++
int main() 
{
    int tables[6][2];

    int count = 0;
    for (int i = 0; i < sizeof(tables) / sizeof(tables[0]); i++)
    {
        scanf("%d%d", &tables[i][0], &tables[i][1]);

        if (tables[i][0] < tables[i][1])
        {
            int temp = tables[i][1];
            tables[i][1] = tables[i][0];
            tables[i][0] = temp;

            count++;
        }
    }

    for (int i = 0; i < sizeof(tables) / sizeof(tables[0]); i++)
        cout << tables[i][0] << tables[i][1] << endl;

    cout << count << "명" << endl;
    return 0;
}
```
