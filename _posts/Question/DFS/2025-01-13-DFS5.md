# 풀이

```c++
#include <iostream>

void DFS(int now);

std::string str;
int map[8][8] = {};

int main()
{
    std::cin >> str;
    for(int i = 0; i < 8; i++)
    {
        for(int j = 0; j < 8; j++)
            std::cin >> map[i][j];
    }

    DFS(0);
    return 0;
}

void DFS(int now)
{
    std::cout << str[now];

    for(int i = 0; i < 8; i++)
    {
        if(map[now][i] == 0)
            continue;

        DFS(i);
    }
}
```
