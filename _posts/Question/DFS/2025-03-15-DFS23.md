# 풀이

```c++
#include <iostream>

using namespace std;

void DFS(int cnt, string meats);

int minCnt = 9999;
bool visited[5];

int main()
{
    string str;
    cin >> str;
    DFS(0, str);
    
    if (minCnt > 4)
        cout << "impossible" << endl;
    else
        cout << minCnt << endl;
    
    return 0;
}

void DFS(int cnt, string meats)
{
    if (meats == "OOOOO" || meats == "XXXXX")
    {
        minCnt = min(cnt, minCnt);
        return;
    }
    
    for (int i = 0; i < 5; i++)
    {
        if (visited[i] == true)
            continue;
         
        visited[i] = true;		
        for (int j = -1; j < 2; j++)
        {
            if (i + j < 0 || i + j >= 5)
                continue;
             
            meats[i + j] = meats[i + j] == 'O' ? 'X' : 'O';
        }

        cout << meats << endl;
        DFS(cnt + 1, meats);
        visited[i] = false;
    }
}
```
