# 풀이

```c++
#include <iostream>
#include <queue>

struct Node
{
    int num;
    int level;
};

std::queue<Node> que;
int map[6][6];

int main()
{
    for (int i = 0; i < 6; i++)
    {
        for (int j = 0; j < 6; j++)
            std::cin >> map[i][j];
    }
        

    que.push(Node{ 0, 0 });

    while (que.empty() == false)
    {
        Node cur = que.front();

        if (cur.num % 2 == 1)
            std::cout << cur.num << " ";

        for (int i = 1; i < 6; i++)
        {
            if (map[cur.num][i] == 1)
                que.push(Node{ i, cur.level + 1 });
        }

        que.pop();
    }

    return 0;
}
```
