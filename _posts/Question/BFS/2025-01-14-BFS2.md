# 풀이

```c++
#include <iostream>
#include <queue>

std::queue<int> que;
int map[8];

int main()
{
    for (int i = 0; i < 8; i++)
        std::cin >> map[i];
    
    que.push(1);
    
    while (que.empty() == false)
    {
        int cur = que.front();
        
        if (map[cur] > 0)
            std::cout << map[cur] << " ";
        
        if(cur * 2 < 8)
            que.push(cur * 2);
        
        if (cur * 2 + 1 < 8)
            que.push(cur * 2 + 1);
        
        que.pop();
    }
    return 0;
}

```
