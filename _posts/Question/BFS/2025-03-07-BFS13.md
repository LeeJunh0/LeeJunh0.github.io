# 풀이

```c++
#include <iostream>
#include <queue>

using namespace std;

int BFS();

int map[4][4] = {};
int direct[4][2] = { -1,0,1,0,0,-1,0,1 };

int main()
{
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
            cin >> map[i][j];
    }
    
    int result = BFS();
    cout << result;
    return 0;
}

int BFS()
{
    queue<pair<int, int>> que;
    que.push({ 0,0 });
    
    int max = 0;
    int cnt = 0;
    
    while (que.empty() == false)
    {
        int check = 0;
        pair<int, int> curPos = que.front();
        que.pop();
        
        if (map[curPos.first][curPos.second] == 1)
        {
            cnt++;
            map[curPos.first][curPos.second] = 0;
        }
        
        for (int i = 0; i < 4; i++)
        {	
            int newY = direct[i][0] + curPos.first;
            int newX = direct[i][1] + curPos.second;
            
            if (newY >= 4 || newX >= 4 || newY < 0 || newX < 0)
                continue;
            
            if (map[newY][newX] == 0)
                continue;
            
            check++;
            cnt++;
            que.push({ newY,newX });
            map[newY][newX] = 0;
        }
        
        if (check <= 0)
            cnt = 0;
        
        if (max < cnt)
            max = cnt;	
    }
    
    return max;
}
```

즉 0,0부터 4방향을 검사해서 섬이 없다면 거기까지가 길이인것이다..

0,0에서 부터 이어지는 섬의길이를 출력하는 문제라는 의미이다.
