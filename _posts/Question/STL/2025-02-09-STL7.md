# 풀이
---------------------------------------
```c++
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>

using namespace std;

int main()
{
    int n;
    cin >> n;
    
    string arrs[3][3];
    
    for (int i = 0; i < n; i++)
    {
        int y, x;
        string str;
        cin >> y >> x >> str;
        
        arrs[y][x] = str;
    }
    
    int windCount;
    cin >> windCount;
    
    for (int i = 0; i < windCount; i++)
    {
        int damage;
        cin >> damage;
        
        for (int j = 0; j < 3; j++)
        {
            for (int k = 0; k < 3; k++)
            {
                if (arrs[j][k].length() <= 0)
                    continue;
                
                if (arrs[j][k].back() - damage <= '0')
                    arrs[j][k].pop_back();
                else
                    arrs[j][k].back() -= damage;
            }
        }
    }
    
    int count = 0;
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            if (arrs[i][j].length() <= 0)
        	      continue;
        
            count += arrs[i][j].length();
        }
    }
    
    cout << count;
    return 0;
}
````
