# 풀이

```c++
void KFC(int* big, int* small);

int main()
{
    int big = 0;
    int small = 0;
    KFC(&big, &small);

    std::cout << "대문자" << big << "개"<<std::endl;
    std::cout << "소문자" << small << "개" << std::endl;

    // 입력 : MiniWorld

    // 결과
    // 대문자2개
    // 소문자7개

    return 0;
}

void KFC(int* big, int* small)
{
    char input[256];
    std::cin >> input;

    for(int i = 0; input[i] != '\0'; i++)
    {
        if(97 <= input[i])
            (*small)++;
        else
            (*big)++;
    }
}
```
