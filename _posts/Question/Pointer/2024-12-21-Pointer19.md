# 풀이

```c++
struct Robot
{
    int a;
    int b;
    char arr[10];
};

int main()
{
    Robot robert;
    std::cin >> robert.a >> robert.b >> robert.arr;

    Robot* p = &robert;
    std::cout << p->a + p->b << " " << p->arr;

    // 입력
    // 3 5 super
    
    // 결과
    // 8 super

    return 0;
}
```
