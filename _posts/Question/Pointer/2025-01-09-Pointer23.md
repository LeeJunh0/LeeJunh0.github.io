# 풀이

```c++
#include <iostream>

struct ABC
{
    int data[4];
    int x;
};

int main()
{
    ABC a, b;
    for (int i = 0; i < 4; i++)
    {
        a.data[i] = i + 1;
        b.data[i] = i + 7;
    }
    a.x = 10;
    b.x = 15;
    
    ABC* p;
    ABC* g;
    
    p = &a;
    g = &b;
    
    for (int i = 0; i < 4; i++)
        std::cout << p->data[i] << " ";

    std::cout << std::endl << p->x;    
    std::cout << std::endl;

    for (int i = 0; i < 4; i++)
        std::cout << g->data[i] << " ";

    std::cout << std::endl << g->x;
    
    return 0;
}
```
