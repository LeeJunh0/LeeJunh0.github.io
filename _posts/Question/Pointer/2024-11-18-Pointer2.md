# 풀이

```c++
int main()
{
    char a, b, c;
    std::cin >> a >> b >> c;
    char* ptrA = &a;
    char* ptrB = &b;
    char* ptrC = &c;

    (*ptrA)++;
    (*ptrB)++;
    (*ptrC)++;

    std::cout << *ptrA << " " << std::endl;
    std::cout << *ptrB << " " << std::endl;
    std::cout << *ptrC << " " << std::endl;

    // 입력 : A B C
    // 결과
    // B C D

    return 0;
}
```
