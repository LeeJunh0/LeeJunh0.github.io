# 풀이

```c++
char arrays[3][3] =
{
    { 'A', 'D', 'F' },
    { 'Q', 'W', 'E' },
    { 'Z', 'X', 'C' }
};

void Find(char input, int* y, int* x);

int main()
{
    char input;
    std::cin >> input;
    int y = 0;
    int x = 0;

    Find(input, &y, &x);
    std::cout << y << ", " << x;

    // 입력 : F

    // 결과
    // 0, 2

    return 0;
}

void Find(char input, int* y, int* x)
{
    for(int i = 0; i < 3; i++)
    {
        *y = i;
        for(int j = 0; j < 3; j++)
        {
            *x = j;
            if(arrays[i][j] == input)
              return;
        }
    }
}
```
