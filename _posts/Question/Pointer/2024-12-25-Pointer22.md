# 풀이

```c++
char str[10];
int data[4];
char* ptrs[4];

int main()
{
    std::cin >> str;
    for(int i = 0; i < 4; i++)
    {
        std::cin >> data[i];
        ptrs[i] = &str[data[i]];
    }

    for(int i = 0; i < 4; i++)
        std::cout << ptrs[i][0];

    // 입력
    // BKABCDE
    // 0 1 5 6

    // 결과
    // BKDE

    return 0;
}
```
