# 풀이

```c++
void Counting();
int IndexOf(char value);
void Print();

char leftMap[4][4];
char rightMap[4][4] =
{
    {'A','B','C','D'},
    {'B','B','A','B'},
    {'C','B','A','C'},
    {'B','A','A','A'}
};

char arr[5] = "ABCD";
int countArr[4] = {};

int main()
{
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
            std::cin >> leftMap[i][j];
    }
    
    Counting();
    Print();

    // 입력
    // ABBA
    // BACB
    // CBAA
    // DDAB

    // 결과
    // B

    return 0;
}

void Counting()
{
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
        {
            if (leftMap[i][j] == rightMap[i][j])
                countArr[IndexOf(leftMap[i][j])]++;
        }
    }
}

int IndexOf(char value)
{
    int idx = 0;
    for (int i = 0; i < 4; i++)
    {
        if (arr[i] == value)
        {
            idx = i;
            break;
        }
    }
    
    return idx;
}

void Print()
{
    int idx = 0;
    int max = 0;
    for (int i = 0; i < 4; i++)
    {
        if (max < countArr[i])
        {
            max = countArr[i];
            idx = i;
        }
    }
    
    std::cout << arr[idx];
}
```
