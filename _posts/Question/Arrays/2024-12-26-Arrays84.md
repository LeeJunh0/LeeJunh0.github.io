# 풀이

```c++
char arr[5][5];

int main()
{
    for (int i = 0; i < 4; i++)
        std::cin >> arr[i];
    
    for (int i = 0; i < 4; i++)
    {
        int count = 0;
        for (int j = 0; j < 4; j++)
        {
            if (arr[j][i] != '#')
                count++;
        }
        std::cout << count << " ";
    }
    
    // 입력
    // #AFG
    // B##E
    // ###D
    // #E##

    // 결과
    // 1 2 1 3

    return 0;
}
```
