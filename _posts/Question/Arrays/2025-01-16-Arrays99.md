# 풀이

```c++
#include <iostream>

void Rotation(int idx);

int gears[3][4] =
{
    3,2,5,3,
    7,6,1,6,
    4,9,2,7
};

int input[4];

int main()
{
    for (int i = 0; i < 4; i++)
        std::cin >> input[i];
    
    for(int i = 0; i < 4; i++)
    {
        for (int j = 0; j < input[i]; j++)
            Rotation(i);
    }
    
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 4; j++)
            std::cout << gears[i][j];
        
        std::cout << std::endl;
    }
    return 0;
}

void Rotation(int idx)
{
    int temp = gears[2][idx];
    
    for (int i = 2; i > 0; i--)
        gears[i][idx] = gears[i - 1][idx];
    
    gears[0][idx] = temp;
}
```
