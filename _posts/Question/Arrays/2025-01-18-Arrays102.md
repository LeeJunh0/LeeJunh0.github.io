# 풀이

```c++
#include <iostream>

void FiveHero();
void DisPatch();

char path[5];
char name[5] = "BIAH";
int visit[4] = {};
int input;
int start;

int main()
{
    std::cin >> input;
    for(int i = 0; i < 4; i++)
    {
        FiveHero();
        DisPatch();
    }

    return 0;
}

void FiveHero()
{
    int count = 0;
    for(int i = start; count < input; i++)
    {
        if(i >= 4)
            i = 0;

        if(visit[i] == 1)
            continue;

        path[count] = name[i];
        count++;
    }
}

void DisPatch()
{
    std::cout << path[input - 1] << " ";

    for(int i = 0; i < 4; i++)
    {
        if(path[input - 1] == name[i])
        {
            visit[i] = 1;
            start = i;
            break;
        }
    }
}
```
