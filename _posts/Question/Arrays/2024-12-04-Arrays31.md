# 풀이

```c++
int main()
{
    int mask1[4][4] = 
    { 
        { 0, 0, 0, 1},
        { 1, 1, 0, 1},
        { 1, 0, 0, 1},
        { 1, 1, 1, 1} 
    };
    int mask2[4][4] =
    {
        { 1, 1, 1, 1},
        { 1, 0, 1, 1},
        { 1, 0, 0, 0},
        { 1, 0, 0, 0}
    };
    
    int arr[4][4];
    
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
        {
            if (mask1[i][j] == 1 || mask2[i][j] == 1)
                arr[i][j] = 1;
            else
                arr[i][j] = 0;
        }
    }
    
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
        {
            if (arr[i][j] == 0)
                std::cout << "(" << i << ", " << j << ")" << std::endl;
        }
    }
    
    // 결과
    // (2, 1)
    // (2, 2)

    return 0;
}
```
