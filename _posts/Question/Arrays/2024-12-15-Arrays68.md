# 풀이

```c++
void Change(int a, int b);
void Print();

int vect[4][3] =
{
    1,5,3,
    4,5,5,
    3,3,5,
    4,6,2
};

int main()
{
    int a, b;
    std::cin >> a >> b;
    
    Change(a, b);
    Print();
    
    // 입력
    // 3 4

    // 결과
    // 1 5 #
    // # 5 5
    // # # 5
    // # 6 2

    return 0;
}

void Change(int a, int b)
{
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            if (a <= vect[i][j] && vect[i][j] <= b)
                vect[i][j] = 0;
        }
    }
}

void Print()
{
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            if (vect[i][j] == 0)
                std::cout << "# ";
            else
                std::cout << vect[i][j] << " ";
        }
        
        std::cout << std::endl;
    }
}
```
