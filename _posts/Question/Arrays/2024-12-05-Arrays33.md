# 풀이

```c++
int Searching();

char vect[] = { 'B','T','K','I','G','Z' };
char target[4];

int main()
{
    for (int i = 0; i < 4; i++)
        std::cin >> target[i];
    
    std::cout << Searching();
    
    // 입력
    // A B Y Z

    // 결과
    // 2

    return 0;
}

int Searching()
{
    int count = 0;
    
    for (int i = 0; i < sizeof(vect) / sizeof(char); i++)
    {
        for (int j = 0; j < 4; j++)
        {
            if (target[j] == vect[i])
                  count++;
        }
    }
    
    return count;
}
```
