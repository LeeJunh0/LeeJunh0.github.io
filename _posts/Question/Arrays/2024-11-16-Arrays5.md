# 풀이

```c++
int GOP();
int SUM();

int main()
{
    int gop = GOP();
    int SUM = SUM();

    if(gop > sum)
        std::cout << "GOP > SUM" << std::endl;
    else if(gop < sum)
        std::cout << "GOP < SUM" << std::endl;
    else
        std::cout << "GOP == SUM" << std::endl;

    // 입력 : 3 6, 11 7
    // 결과
    // GOP == SUM
    return 0;
}

int GOP()
{
    int a, b;
    scanf("%d %d", &a, &b);
    return a * b;
}

int SUM()
{
    int a, b;
    scanf("%d %d", &a, &b);
    return a + b;
}
```
