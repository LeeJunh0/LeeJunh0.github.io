# 풀이

```c++
#include <iostream>

int n;	
int arr[5];

int main()
{
    std::cin >> n;
    for (int i = 0; i < n; i++)
        std::cin >> arr[i];
    
    int result[5] = {};
    for (int i = 0; i < n; i++)
    {
        int min = 99;
        int idx = i;
        for (int j = i; j < n; j++)
        {
            if (min > arr[j])
            {
                min = arr[j];
                idx = j;
            }
        }
        
        int temp = arr[idx];
        arr[idx] = arr[i];
        arr[i] = temp;
        result[i] = arr[i];
    }
    
    for (int i = 0; i < 3; i++)
    {
        if (result[i] == 0 && i == 0)
        {
            for (int j = 1; j < 5; j++)
            {
                if (result[j] <= 0)
                    continue;
                
                int temp = result[i];
                result[i] = result[j];
                result[j] = temp;
                break;
            }
        }
        
        std::cout << result[i];
    }
    
    return 0;
}
```
