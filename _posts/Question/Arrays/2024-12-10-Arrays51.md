# 풀이

```c++
void Func(int input, int* cur);
void Print();

int vect[4][4] = {};

int main()
{
    int cur = 1;
    for (int i = 0; i < 4; i++)
    {
        int input;
        std::cin >> input;
        
        Func(input, &cur);
        cur++;
    }
    
    Print();
    
    return 0;
}

void Func(int input, int* cur)
{
    int num = 1;
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
        {
            if (num == input)
            {
                vect[i][j] = *cur;
                return;
            }
            
            num++;
        }
    }
}

void Print()
{
    for (int i = 0; i < 4; i++)
    {
        for (int j = 0; j < 4; j++)
            std::cout << vect[i][j];
        
        std::cout << std::endl;
    }
}
```
