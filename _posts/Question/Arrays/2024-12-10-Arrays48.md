# 풀이

```c++
void Change();
void Print();

int input[4][2];
int vect[4][3] = {};

int main()
{
    for (int i = 0; i < 4; i++)
        std::cin >> input[i][0] >> input[i][1];
    
    Change();
    Print();
    
    // 입력
    // 0 1
    // 1 0
    // 3 0
    // 3 1

    // 결과
    // 0 5 0
    // 5 0 0
    // 0 0 0
    // 5 5 0

    return 0;
}

void Change()
{
    for (int i = 0; i < 4; i++)
        vect[input[i][0]][input[i][1]] = 5;
}

void Print()
{
    for(int i = 0; i < 4; i++)
    {
        for(int j = 0; j < 3; j++)
        std::cout << vect[i][j] << " ";
    
        std::cout << std::endl;
    }
}
```
---------------------------------------
