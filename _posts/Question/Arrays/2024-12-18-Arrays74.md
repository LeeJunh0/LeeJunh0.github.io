# 풀이

```c++
char third[4][5][3] =
{
    {
        ' ','#',' ',
        '#',' ','#',
        '#','#','#',
        '#',' ','#',
        '#',' ','#'
    },
    {
        '#','#','#',
        '#',' ','#',
        '#','#','#',
        '#',' ','#',
        '#','#','#'
    },
    {
        '#','#','#',
        '#',' ','#',
        '#',' ',' ',
        '#',' ','#',
        '#','#','#'
    },
    {
        '#','#',' ',
        '#',' ','#',
        '#',' ','#',
        '#',' ','#',
        '#','#',' '
    }
};

int main()
{
    int input;
    std::cin >> input;
    
    for (int i = 0; i < 5; i++)
    {
        for (int j = 0; j < 3; j++)
            std::cout << third[input][i][j];
        
        std::cout << std::endl;
    }
    
    // 입력
    // 1

    // 결과
    // ###
    // # #
    // ###
    // # #
    // ###


    return 0;
}
```
