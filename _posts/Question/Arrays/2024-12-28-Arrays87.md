# 풀이

```c++
char vect[3][5];

int main()
{
    std::string str;
    std::cin >> str;

    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < 5; j++)
            vect[i][j] = str[i * 5 + j];
    }

    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < 5; j++)
            std::cout << vect[i][j];

        std::cout << std::endl;
    }

    // 입력
    // 011011010101101

    // 결과
    // 01101
    // 10101
    // 01101

    return 0;
}
```
