이건 최댓값 최솟값을 찾는 문제다.

# 문제   
---------------------------------------

![Desktop View](/assets/img/Array15.png){: w="700" h="400" }

---------------------------------------

# 풀이

```c++
int main()
{
    int arr[7];
    
    for(int i = 0; i < 7; i++)
        std::cin >> arr[i];

    int curMax = -99999;
    int curMin = 99999;

    for(int i = 0; i < 7; i++)
    {
        if(curMax > arr[i])
            curMax = arr[i];

        if(curMin < arr[i])
            curMin = arr[i];
    }

    std::cout << "MAX = " << curMax << std::endl;
    std::cout << "MIN = " << curMin << std::endl;

    // 입력 : 4 1 5 2 3 2 2

    // 결과
    // MAX = 5
    // MIN = 1

    return 0;
}
```
