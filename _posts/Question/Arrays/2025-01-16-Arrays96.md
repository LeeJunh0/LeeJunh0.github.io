# 풀이

```c++
#include<iostream>

int arr[3][3];

int main()
{
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
            std::cin >> arr[i][j];
    }
    
    for (int i = 0; i < 3; i++)
    {
        int num = arr[i][0];
        bool isCheck = true;
        for (int j = 0; j < 3; j++)
        {
            if (num == arr[i][j])
                continue;
            
            isCheck = false;
            break;
        }
    
        if (isCheck == true)
            std::cout << num << std::endl;
        else
            std::cout << "x" << std::endl;
    }
    return 0;
}
```
