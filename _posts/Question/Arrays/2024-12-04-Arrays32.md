# 풀이

```c++
int mask[2][5] = 
{ 
    { 0, 0, 1, 0, 0 },
    { 0, 0, 1, 1, 1 } 
};

int arrays[2][5] = 
{ 
    { 3, 5, 4, 1, 1 },
    { 3, 5, 2, 5, 6 } 
};

void Searching(int input);

int main()
{
    int input;
    std::cin >> input;
    Searching(input);
    
    // 입력
    // 1

    // 결과
    // 1 없음

    return 0;
}

void Searching(int input)
{
    for (int i = 0; i < 2; i++)
    {
        for (int j = 0; j < 5; j++)
        {
            if (mask[i][j] == 1)
            {
                if (arrays[i][j] == input)
                {
                    std::cout << input << " 있음";
                    return;
                }
            	
            }
        }
    }
    
    std::cout << input << " 없음";
}
```
