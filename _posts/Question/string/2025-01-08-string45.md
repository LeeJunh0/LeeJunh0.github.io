# 풀이

```c++
#include <iostream>

void Command();
void Left();
void Right();
void Delete();

std::string str;
std::string cmd;
int cursorIdx;

int main()
{
    std::cin >> str >> cursorIdx >> cmd;
    Command();
    std::cout << cursorIdx;
    return 0;
}

void Command()
{
	for (int i = 0; i < cmd.length(); i++)
	{
	    if (cmd[i] == 'L')
	        Left();
	    else if (cmd[i] == 'R')
	        Right();
	    else
	        Delete();
	}
}

void Left() { cursorIdx--; }
void Right() {  cursorIdx++; }
void Delete()
{
    for (int i = cursorIdx; i < str.length(); i++)	
        str[i] = str[i + 1];
}

```
