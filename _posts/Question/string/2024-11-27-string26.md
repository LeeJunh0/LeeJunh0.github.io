# 풀이

```c++
void Process(char* str, int index);

int main()
{
    char str[256];
    int input;
    std::cin >> str >> input;

    Process(str, input);

    // 입력
    // gtkbacd
    // 3

    // 결과
    // gtkacd

    return 0;
}

void Process(char* str, int index)
{
    int length = 0;
    for(int i = 0; str[i] != '\0'; i++)
        length++;

    char arr[256];
    int idx = 0;
    for(int i = 0; i < length + 1; i++)
    {
        if(index == i)
            continue;

        arr[idx] = str[i];
        idx++;
    }

    std::cout << arr;
}


```
---------------------------------------
