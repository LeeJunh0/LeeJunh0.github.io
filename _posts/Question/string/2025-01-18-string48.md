# 풀이

```c++
#include <iostream>

std::string strs[3];

int main()
{
    for(int i = 0; i < 3; i++)
        std::cin >> strs[i];

    std::string maxStr = strs[0];
    for(int i = 1; i < 3; i++)
    {
        if(maxStr.length() < strs[i].length())
            maxStr = strs[i];

        else if(maxStr.length() == strs[i].length())
        {
            for(int j = 0; j < maxStr.length(); j++)
            {
                if(maxStr[j] >= strs[i][j])
                    continue;

                maxStr = strs[i];
            }
        }
    }

    std::cout << maxStr;
    return 0;
}
```
