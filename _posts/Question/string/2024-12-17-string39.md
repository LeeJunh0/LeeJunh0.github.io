# 풀이

```c++
int main()
{
    int floor = 0;
    for (int i = 0; i < 5; i++)
    {
        char input[5] = "";
        std::cin >> input;
        
        if (strcmp(input, "up") == 0)
            floor++;
        else
            floor--;
    }
    
    if (floor > 0)
        std::cout << floor;
    else
        std::cout << "B" << floor * -1;
    
    // 입력
    // up
    // down
    // down
    // down
    // down

    // 결과
    // B3

    return 0;
}
```
