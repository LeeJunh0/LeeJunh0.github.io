# 풀이

```c++
void AddChar(char* str, int index);

int main()
{
    char str[6];
    int input;
    std::cin >> str >> input;

    AddChar(str, input);

    // 입력
    // show
    // 2

    // 결과
    // shAow

    return 0;
}

void AddChar(char* str, int index)
{
    char arr[6]

    int length = 0;
    for(int i = 0; str[i] != '\0'; i++)
        length++;

    int idx = 0;
    for(int i = 0; i < length; i++)
    {
        if(index == idx)
        {
            arr[i] = index;
            idx++;
        }

        arr[idx] = str[i];
        idx++;
    }

    arr[length + 1] = '\0';
    std::cout << arr;
}
```
