# 풀이

```c++
int main()
{
    char arr1[256];
    char arr2[256];
    std::cin >> arr1 >> arr2;

    int count1 = 0;
    int count2 = 0;
    for(int i = 0; i < 256; i++)
    {
        if(arr1[i] != '\0')
            count1++;
    }

    for(int i = 0; i < 256; i++)
    {
        if(arr2[i] != '\0')
            count2++;
    }

    std::cout << count1 << " " << count2;

    // 입력 ABCD BBQ
    // 결과
    // 4 3

    return 0;
}

```
