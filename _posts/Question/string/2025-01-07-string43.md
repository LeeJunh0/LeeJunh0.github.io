# 풀이

```c++
#include <iostream>

int main()
{
    std::string str;
    std::cin >> str;
    
    int idx = 0;
    int arr[10];
    int result = 0;
    for (int i = 0; i < str.length(); i++)
    {
        if (str[i] > 47 && str[i] < 58)
        {
            arr[idx] = str[i] - 48;
            idx++;
        }
    }
    
    int digit = 1;
    for (int i = idx - 1; i >= 0; i--)
    {
        result += arr[i] * digit;
        digit *= 10;
    }
    std::cout << result + 5;
    
    return 0;
}
```
