# 풀이

```c++
#include <iostream>
#include <cstring>

char strings[3][11];

int MaxLength();
void Swap(int idx);

int main()
{
    for (int i = 0; i < 3; i++)
        std::cin >> strings[i];
    
    int max = MaxLength();
    Swap(max);
    
    for (int i = 0; i < 3; i++)
        std::cout << strings[i] << std::endl;
    
    // 입력
    // KFC
    // ABCDEFG
    // BBQ

    // 결과
    // ABCDEFG
    // KFC
    // BBQ

    return 0;
}

int MaxLength()
{
    int max = 0;
    int idx = 0;
    for (int i = 0; i < 3; i++)
    {	
        int result = strlen(strings[i]);
        if (max < result)
        {
            max = result;
            idx = i;
        }
    }
    
    return idx;
}

void Swap(int idx)
{
    char temp[11];
    strcpy_s(temp, strings[idx]);
    strcpy_s(strings[idx], strings[0]);
    strcpy_s(strings[0], temp);
}
```
