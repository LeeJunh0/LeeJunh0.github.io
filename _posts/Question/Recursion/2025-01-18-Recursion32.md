# 풀이

```c++
#include <iostream>

void Recursion(int level);

char arr[] = "ox";
char path[10] = "";
int depth;

int main()
{
    std::cin >> depth;
    Recursion(0);

    return 0;
}

void Recursion(int level)
{
    if(level >= depth)
    {
        std::cout << path << std::endl;
        return;
    }

    for(int i = 0; i < 2; i++)
    {
        path[level] = arr[i];
        Recursion(level + 1);
    }
}
```
