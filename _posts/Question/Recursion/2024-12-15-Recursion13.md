# 풀이

```c++
void Recursion(int n);

int level, branch, count = 1;

int main()
{
    std::cin >> branch >>level;
    Recursion(0);
    std::cout << count;

    // 입력
    // 4 3

    // 결과
    // 85

    return 0;
}

void Recursion(int n)
{
    if(n == level)
        return;

    for(int i = 0; i < branch; i++)
    {
        count++;
        Recursion(n + 1);
    }
}
```
