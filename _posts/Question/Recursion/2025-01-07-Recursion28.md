# 풀이

```c++
void Recursion(int level);

char path[5];
char card[] = { 'A','B','C','D' };
int visit[4];
int input;

int main()
{
    std::cin >> input;
    
    Recursion(0);
    return 0;
}

void Recursion(int level)
{
    if (level == input)
    {
        std::cout << path << std::endl;
        return;
    }
    
    for (int i = 0; i < strlen(card); i++)
    {
        if (visit[i] == 1)
            continue;
        
        visit[i] = 1;
        path[level] = card[i];
        Recursion(level + 1);
        path[level] = 0;
        visit[i] = 0;
    }
}
```
