# 풀이

```c++
#include <iostream>

void Recursion(int idx);

int map[10] = { 0,3,1,2,1,3,2,1,2,1 };

int main()
{
    int input;
    std::cin >> input;
    
    std::cout << "시작 ";
    Recursion(input);
    std::cout << "시작 ";
    return 0;
}

void Recursion(int idx)
{
    if (idx >= 10)
    {
        std::cout << "도착 ";
        return;
    }
    
    std::cout << map[idx] << " ";
    Recursion(idx + map[idx]);
    std::cout << map[idx] << " ";
}
```
