# 풀이

```c++
#include <iostream>
#include <queue>

std::queue<int> que;

int main()
{
    que.push(1);
    int input;
    std::cin >> input;
    
    int result = 0;
    for (int i = 0; i <= input; i++)
    {
        result += que.front();
        que.push(que.front() * 3);
        que.pop();
    }
    
    std::cout << result;
    return 0;
}
```
