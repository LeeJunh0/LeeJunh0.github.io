# 풀이

```c++
void Recursion(int level);

int input;
int cur = 0;
int count = 0;
char name[4] = "ABC";
char path[4];

int main()
{
    std::cin >> input;
    
    Recursion(0);
    std::cout << count;

    // 입력
    // 3

    // 결과
    // 24

    return 0;
}

void Recursion(int level)
{
    if (level == 3)
    {
        for (int i = 1; i < 3; i++)
        {
            if (path[0] == path[i])
                cur++;
        }
        
        if (cur < 2)
        {
            //std::cout << path << std::endl;
            count++;
        }
        
        cur = 0;
        return;
    }
    
    for (int i = 0; i < 3; i++)
    {
        path[level] = name[i];
        Recursion(level + 1);
        path[level] = 0;
    }
}
```
