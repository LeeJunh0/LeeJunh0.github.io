# 풀이

```c++
void Recursion(int level);

char path[5] = "";
char name[5] = "BGTK";

int depth;

int main()
{
    std::cin >> depth;
    Recursion(0);
    
    // 입력
    // 2

    // 결과
    // BB
    // BG
    // BT
    // BK
    // GB
    // GG
    // GT
    // GK
    // TB
    // TG
    // TT
    // TK
    // KB
    // KG
    // KT
    // KK

    return 0;
}

void Recursion(int level)
{
    if (level == depth)
    {
        std::cout << path << std::endl;
        return;
    }
    
    for (int i = 0; i < 4; i++)
    {
        path[level] = name[i];
        Recursion(level + 1);
        path[level] = 0;
    }
}
```
