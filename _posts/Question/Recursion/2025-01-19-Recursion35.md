# 풀이

```c++
#include <iostream>

void Recursion(int level);

std::string mans[3];
std::string path[3];

int main()
{
    for (int i = 0; i < 3; i++)
        std::cin >> mans[i];
    
    Recursion(0);
    return 0;
}

void Recursion(int level)
{
    if (level >= 3)
    {
        for (int i = 0; i < 3; i++)
            std::cout << path[i] << " ";
        
        std::cout << std::endl;
        return;
    }
    
    for (int i = 0; i < 3; i++)
    {
        path[level] = mans[i];
        Recursion(level + 1);
        path[level] = "";
    }
}
```
