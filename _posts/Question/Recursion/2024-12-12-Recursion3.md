# 풀이

```c++
int arr[6];

void Recursion(int i);

int main()
{
    for (int i = 0; i < 6; i++)
        std::cin >> arr[i];
     
    Recursion(0);
     
    return 0;
}

void Recursion(int i)
{
    std::cout << arr[i] << " ";
    if (i == 5)
        return;
    
    Recursion(i + 1);
    std::cout << arr[i] << " ";
}
```
