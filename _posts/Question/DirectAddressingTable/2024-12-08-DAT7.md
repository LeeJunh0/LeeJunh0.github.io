# 풀이

```c++
int train[] = { 3,7,6,4,2,9,1,7 };
int team[3];

bool Pattern(int idx);

int main()
{
    for(int i = 0; i < 3; i++)
        std::cin >> team[i];

    for(int i = 0; i < 8; i++)
    {
        if(Pattern(i) == true)
            std::cout << i << "번~" << i + 2 << "번 칸";
    }

    // 입력
    // 9 1 7

    // 결과
    // 5번~7번 칸

    return 0;
}

bool Pattern(int idx)
{
    for(int i = 0; i < 3; i++)
    {
        if(train[idx + i] != team[i])
            return false;
    }

    return true;
}
```
