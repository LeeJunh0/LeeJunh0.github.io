# 풀이

```c++
int arr[256] = {};
char str[9];

char Max();

int main()
{
    std::cin >> str;
    for (int i = 0; i < 9; i++)
        arr[str[i]]++;
    
    std::cout << Max();
    
    // 입력
    // BQTABAAA

    // 결과
    // A

    return 0;
}

char Max()
{
    int max = 0;
    int idx = 0;
    for (int i = 0; i < 256; i++)
    {
        if (max < arr[i])
        {
            max = arr[i];
            idx = i;
        }
    }
    
    return (char)idx;
}
```
