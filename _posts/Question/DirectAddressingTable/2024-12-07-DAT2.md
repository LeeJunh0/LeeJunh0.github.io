# 풀이

```c++
void Check(int* arr, int(*list)[4]);
int Searching(int* arr);

int main()
{
    int Id[65536] = {};
    int list[3][4] = 
    {
        { 65000,35,42,70 },
        { 70,35,65000,1300 },
        { 65000,30000,38,42 }
    };

    Check(Id, list);
    std::cout << Searching(Id);

    // 결과
    // 65000

    return 0;
}

void Check(int* arr, int(*list)[4])
{
    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < 4; j++)
            arr[list[i][j]]++;
    }
}

int Searching(int* arr)
{
    int max = 0;
    int idx = 0;
    for(int i = 0; i < 65536; i++)
    {
        if(max < arr[i])
        {
            max = arr[i];
            idx = i;
        }+
    }

    return idx;
}
```
