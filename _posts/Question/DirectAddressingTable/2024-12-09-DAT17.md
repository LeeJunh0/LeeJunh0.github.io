# 풀이

```c++
bool Pattern(int start, int length);
void Print(bool flag);

char input[256];

int main()
{
    std::cin >> input;
    
    int length = 1;
    for (int i = 0; input[i] != '\0'; i++)
        length++;
    
    bool flag = false;
    for (int i = 0; i < length - 5; i++)
    {
        flag = Pattern(i, length);
        
        if (flag == true)
            break;
    }
    
    Print(flag);
    
    // 입력
    // QGHOSTA

    // 결과
    // 존재
    
    return 0;
}

bool Pattern(int start, int length)
{
    char pattern[] = { 'G','H','O','S','T' };
    for (int i = 0; i < 5; i++)
    {
        if (input[start + i] != pattern[i])
            return false;
    }
    
    return true;
}

void Print(bool flag)
{
    if (flag == true)
        std::cout << "존재";
    else
        std::cout << "존재하지 않음";
}
```
