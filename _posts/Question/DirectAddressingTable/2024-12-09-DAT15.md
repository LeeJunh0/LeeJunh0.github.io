# 풀이

```c++
void Delete();
void Print();

int Dat[256] = {};
char input[11];

int main()
{
    std::cin >> input;
    
    Delete();
    Print();
    
    // 입력
    // ABBACCDEA

    // 결과
    // ABCDE

    return 0;
}

void Delete()
{
    for (int i = 0; input[i] != '\0'; i++)
        Dat[input[i]]++;
}

void Print()
{
    for(int i = 0; i < 256; i++)
    {
        if (Dat[i] >= 1)
            std::cout << (char)i;
    }
}
```
