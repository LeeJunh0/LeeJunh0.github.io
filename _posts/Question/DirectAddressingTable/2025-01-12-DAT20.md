# 풀이

```c++
#include <iostream>

int arr[4][3] =
{
	3,5,1,
	3,1,2,
	3,4,6,
	5,4,6
};
int dat[7] = {};

int main()
{
	for (int i = 0; i < 4; i++)
	{
	    for (int j = 0; j < 3; j++)
	        dat[arr[i][j]]++;
	}

	for(int i = 1; i < 7; i++)
	{
	    std::cout << i << " ";
	    for (int j = 0; j < dat[i]; j++)
	        std::cout << "*" << " ";

	    std::cout << std::endl;
	}

	return 0;
}
```
