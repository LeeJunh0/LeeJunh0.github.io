# 풀이

```c++
bool Searching(char word);

int Dat[256] = {};
char vect[7] = "CODING";

int main()
{
    int n;
    std::cin >> n;
    
    for (int i = 0; i < 7; i++)
        Dat[vect[i]]++;
    
    for (int i = 0; i < n; i++)
    {
        char input;
        std::cin >> input;
        bool flag = Searching(input);
        
        if (flag == true)
            std::cout << "O";
        else
            std::cout << "X";
    }
    
    // 입력
    // 3
    // I N G

    // 결과
    // OOO

    return 0;
}

bool Searching(char word)
{
    if (Dat[word] == 1)
        return true;
    else
        return false;
}
```
