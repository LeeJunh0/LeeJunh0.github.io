# 풀이

```c++
void Check(int* arr, int(*vect)[3]);

int main()
{
    int arr[10] = {1, };
    int vect[3][3];
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
            std::cin >> vect[i][j];
    }
    
    Check(arr, vect);
    
    // 입력
    // 3 5 5
    // 1 4 9
    // 2 2 1

    // 결과
    // 6 7 8

    return 0;
}

void Check(int* arr, int(*vect)[3])
{
    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < 3; j++)
            arr[vect[i][j]] = 1;
    }

    for(int i = 0; i < 10; i++)
    {
        if(arr[i] == 0)
            std::cout << i << " ";
    }
}
```
