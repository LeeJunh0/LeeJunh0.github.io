# 풀이

```c++
bool Pattern(int* floor);

int apart[5][3] = 
{
    { 15,18,17 },
    { 4,6,9 },
    { 10,1,3 },
    { 7,8,9 },
    { 15,2,6 }
};
int family[3];

int main()
{
    for(int i = 0; i < 3; i++)
        std::cin >> family[i];

    int floor = 1;
    for(int i = 4; i >= 0; i--)
    {
        if(Pattern(apart[i]) == true)
        {
            std::cout << floor << "층";
            break;
        }

        floor++;
    }

    return 0;
}

bool Pattern(int* floor)
{
    for(int i = 0; i < 3; i++)
    {
        if(floor[i] != family[i])
            return false;
    }

    return true;
}
```
