# 풀이

```c++
int Dat[256] = {};
char vect[5][3] =
{
    { 'A','B','C' },
    { 'A','G','H' },
    { 'H','I','J' },
    { 'K','A','B' },
    { 'A','B','C' }
};

void Sort();
void Print();

int main()
{
    Sort();
    Print();

    // 결과
    // AAAABBBCCGHHIJK

    return 0;
}

void Sort()
{
    for (int i = 0; i < 5; i++)
    {
        for (int j = 0; j < 3; j++)        
            Dat[vect[i][j]]++;        
    }
}

void Print()
{
    for (int i = 0; i < 256; i++)
    {
        for (int j = 0; j < Dat[i]; j++)		
            std::cout << (char)i;
    }
}
```
