# 풀이

```c++
int Check();

int Dat[256] = {};
char town[3][3] =
{
    { 'C','D','A' },
    { 'B','M','Z' },
    { 'Q','P','O' }
};
char blackList[4];

int main()
{
    for (int i = 0; i < 4; i++)
        std::cin >> blackList[i];
    
    std::cout << Check() << "명";
    
    // 입력
    // BTCK

    // 결과
    // 2명

    return 0;
}

int Check()
{
    int count = 0;
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
            Dat[town[i][j]] = 1;
    }
    
    for (int i = 0; i < 4; i++)
    {
        if (Dat[blackList[i]] == 1)
            count++;
    }
    
    return count;
}
```
