# 풀이

```c++
void Counting();

int Dat[10] = {};
int vect[3][5] =
{
    { 1,3,3,5,1 },
    { 3,6,2,4,2 },
    { 1,9,2,6,5 }
};

int main()
{
    int input;
    std::cin >> input;

    Counting();

    for(int i = 0; i < 10; i++)
    {
        if(input == Dat[i])
            std::cout << i <<" ";
    }

    // 입력
    // 3

    // 결과
    // 1 2 3

    return 0;
}

void Counting()
{
    for(int i = 0; i < 3; i++)
    {
        for(int j = 0; j < 5; j++)
            Dat[vect[i][j]]++;
    }
}
```
