# 풀이

```c++
void Output(char* arr);

int main()
{
    char arr[11];
    std::cin >> arr;
    
    Output(arr);
    
    // 입력
    // AgKeOqv

    // 결과
    // 일반문장

    return 0;
}

void Output(char* arr)
{
    bool isCheck = true;
    for (int i = 0; arr[i] != '\0'; i++)
    {
        if (i % 2 == 1) 
        {
            if (97 <= arr[i])
                continue;
            else
            {
                isCheck = false;
                break;
            }
        }
        else 
        {
            if (65 <= arr[i] && 90 >= arr[i])
                continue;
            else
            {
                isCheck = false;
                break;
            }
        }
    }
    
    if (isCheck == true)
        std::cout << "개구리문장";
    else
        std::cout << "일반문장";
}
```
