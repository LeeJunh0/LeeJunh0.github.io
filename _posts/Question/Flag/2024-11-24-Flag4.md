# 풀이

```c++
int main()
{
    char vect[4][5];
    int colum = sizeof(vect) / sizeof(vect[0]);
    
    for (int i = 0; i < colum; i++)
        std::cin >> vect[i];
    
    bool flagA = false;
    bool flagB = false;
    for (int i = 0; i < colum; i++)
    {
        for (int j = 0; vect[i][j] != '\0'; j++)
        {
            if (vect[i][j] == 'A')
                flagA = true;
    
            if (vect[i][j] == 'B')
                flagB = true;
        }
    }
    
    if (flagA == true && flagB == true)
        std::cout << "대발견";
    else if (flagA == true || flagB == true)
        std::cout << "중발견";
    else
        std::cout << "미발견";
    
    // 입력
    // SHOW
    // YOUR
    // JASON
    // DATA

    // 결과
    // 중발견

    return 0;
}
```
