# 풀이

```c++
int main()
{
    char arr[20];
    std::cin >> arr;

    int length = 0;
    for(int i = 0; arr[i] != '\0'; i++)
        length++;


    // 버블정렬로 한 풀이.
    // for(int i = length - 1; i >= 0; i--)
    // {
    //     for(int j = 0; j < length; j++)
    //     {
    //         if(arr[j] > arr[j + 1])
    //         {
    //             int temp = arr[j];
    //             arr[j] = arr[j + 1];
    //             arr[j + 1] = temp;
    //         }
    //     }
    // }

    int minValue;
    int minIndex;
    for(int i = 0; i < length; i++)
    {
        minValue = arr[i];
        minIndex = i;
        for(int j = i; j < length; j++)
        {
            if(minValue > arr[j])
            {
                minValue = arr[j];
                minIndex = j;
            }
        }

        int temp = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = temp;
    }

    std::cout << arr;

    // 입력
    // ASDASDASD

    // 결과
    // AAADDDSSS

    return 0;
}
```
