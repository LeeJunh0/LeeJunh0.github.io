# 풀이

```c++
#include <iostream>

struct Node
{
    int a;
    char b;
    
    void pop();
};

Node que[10];
int size = 0;

int main()
{
    int input;
    std::cin >> input;
    
    for(int i = 0; i < input; i++)
    {
        std::cin >> que[i].a >> que[i].b;
        size++;
    }
    
    for (int i = 0; i < input; i++)
        que->pop();
    
    return 0;
}

void Node::pop()
{
    if (size == 0)
        return;
    
    std::cout << que[0].a << " " << que[0].b << std::endl;
    
    for (int i = 0; i < size; i++)
        que[i] = que[i + 1];
    
    size--;
}
```
